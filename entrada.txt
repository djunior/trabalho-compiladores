inc integer x: integer {
	local integer a = 10;
	local integer b = 5;
	<= x + a + b;
}

decrem integer x: integer {
	<= x - 1;
}

sum integer x, integer y: integer{
    <= x + y;
}

mdc integer a, integer b: integer
{
  local integer resto;
  while b > 0 :
  {
    resto = a % b;
    a = b;
    b = resto;
  }
  <= a;
}
